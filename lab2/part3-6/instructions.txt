-- installing sequelize

# Use npm on the commandline:
npm install sequelize

// Then require the installed library in your application code:
var Sequelize = require("sequelize")


-- install pg

npm install --save pg


-- create new database

CREATE DATABASE lab2;


-- install migration tool

npm install --save sequelize-cli

npm install -g sequelize-cli


-- init the sequelize directory

sequelize init



-- make database readable by sequelize

sequelize model:create --name Judge --attributes "id:integer, name:string, room:integer, ext:string"
sequelize model:create --name Courtroom --attributes "id:integer, , number:string"
sequelize model:create --name Participant --attributes "id:integer, , name:string, address:string, type:string"
sequelize model:create --name Case --attributes "judge_id:integer, courtroom_id:integer, claimant_id:integer, respondent_id:integer, start_date:date, duration:integer, result:boolean"

sequelize db:migrate
